消息格式

雷达板中面我设置的消息传递的格式为消息头+消息内容，因此在上位机读取消息的时候，需要将消息头忽略，只需要用到消息体，下面我将直接给出代码

代码：

classdef adcdata_app < matlab.apps.AppBase

    % Properties that correspond to app components
    properties (Access = public)
        UIFigure           matlab.ui.Figure
        dataPanel          matlab.ui.container.Panel
        originDataAxes     matlab.ui.control.UIAxes
        fftDataAxes        matlab.ui.control.UIAxes
        uartPanel          matlab.ui.container.Panel
        Label              matlab.ui.control.Label
        uartPort_DropDown  matlab.ui.control.DropDown
        Label_2            matlab.ui.control.Label
        bandRate_DropDown  matlab.ui.control.DropDown
        uartOpenButton     matlab.ui.control.Button
        uartCloseButton    matlab.ui.control.Button
    end


    properties (Access = private)
        COM                         % port
        baudRate                    % baudRate
        SERIAL                      % SERIAL
        chirpFlag                   % chirp flag
        recvData_once               % once receive bytes
        recvData_total              % total receive bytes
        samples                     % samples 
        samplesIndex                % samples index
        c                           % light speed
        fc                          % center frequency
        lambda                      % wave length
        chirpSlope                  % chirp slope
        rangeIndex                  % range Index
        fs                          % frequency samples
        freq_bin                    % frequency bin
    end

    methods (Access = private)

        function results = serial_readCallback(app,obj,event)
            %get data from serial
            data = getSreialData(app,obj);
            %get data's bytes
            [~,data_n] = size(data);

            if data_n > 1000
                %find the chirpFlag array
                arrayIndex = strfind(data,app.chirpFlag);

                %arrayIndex array size
                [~,arrayIndexn] = size(arrayIndex);

                %find first data location
                data = data(arrayIndex(1):end);

                %update arrayIndex
                arrayIndex = arrayIndex - arrayIndex(1) + 1;

                %loops
                while arrayIndexn > 1

                    %get chirp data
                    recvdata16 = data(6:arrayIndex(2)-arrayIndex(1));

                    %allocate memory
                    cmplxArray = zeros(1,app.samples);
%                     fft_data = zeros(1,app.samples);

                    %get samples
                    for j = 1:app.samples
                        real1 = recvdata16(2*j-1);  %get real from serial
                        imag1 = recvdata16(2*j);    %get imag from serial
                        cmplxArray(1,j) = complex(real1,imag1);     %get complex data
                    end

                    %generate window
                    range_win = hamming(app.samples); 

                    %windowing
                    cmplxArray_win = cmplxArray .* range_win';

                    %fft
                    fft_data = fft(cmplxArray_win);
                    fft_data = abs(fliplr(fft_data));

                    plot_data(app,cmplxArray,fft_data);

                    %update arrayIndex data
                    arrayIndex = arrayIndex(2:end);
                    %update data
                    data = data(arrayIndex(1):end);
                    %updata arrayIndex
                    arrayIndex = arrayIndex - arrayIndex(1) + 1;
                    %get data's bytes
                    [~,arrayIndexn] = size(arrayIndex);
                end

                %remove processed data
                data = data(arrayIndex(1):end);

                %update total data
                app.recvData_total = data;
            end
%             results = none;
        end

        function results = getSreialData(app,obj)

            numBytesAvail = obj.NumBytesAvailable;  % get available numbers of bytes

            if numBytesAvail > 0

                %read data from serial
                app.recvData_once = read(obj,numBytesAvail,"int16");

                %add the currently read content to the total content
                app.recvData_total = horzcat(app.recvData_total,app.recvData_once); 
            end

            results = app.recvData_total;
        end

        function results = plot_data(app,original_data,fft_data)

            plot(app.originDataAxes,app.samplesIndex,real(original_data));
            hold(app.originDataAxes,"on");
            plot(app.originDataAxes,app.samplesIndex,imag(original_data));
            hold(app.originDataAxes,"off");

            plot(app.fftDataAxes,app.rangeIndex,fft_data);

%             results = none;
        end
    end


    % Callbacks that handle component events
    methods (Access = private)

        % Code that executes after component creation
        function startupFcn(app)
            try
                fclose(instrfind);
            catch 

            end
            Ports = instrhwinfo('serial');  %find available port
            app.uartPort_DropDown.Items = flipud(Ports.SerialPorts); %port number assignment dropdown
            app.bandRate_DropDown.Items = {'115200','961200'}; %bandrate assignment dropdown

            app.chirpFlag = [4369 8738 13107 8738 4369]; %first chirp flag

            app.samples = 256; %samples
            app.samplesIndex = 1:1:app.samples; %samples Index

            app.c = 3e8;            %unit:m/s
            app.fc = 77e9;
            app.lambda = app.c / app.fc;
            app.chirpSlope = 12.022e12;   %unit:Hz/s
            app.fs = 2.0e6;

            app.freq_bin = (app.samplesIndex - 1) * app.fs / app.samples;
            app.rangeIndex = app.freq_bin * app.c / 2 / app.chirpSlope;
            app.originDataAxes.XLim = [app.samplesIndex(1) app.samplesIndex(app.samples)];
            app.fftDataAxes.XLim = [app.rangeIndex(1) app.rangeIndex(app.samples)];
        end

        % Button pushed function: uartOpenButton
        function uartOpenButtonPushed(app, event)
            %open UART
            try
                app.COM = app.uartPort_DropDown.Value;   %get COM port
                app.baudRate = str2double(app.bandRate_DropDown.Value);  %baudrate
                app.SERIAL = serialport(app.COM,app.baudRate);  %create serialport object
                configureCallback(app.SERIAL,"byte",2^11,@app.serial_readCallback);    %configure callback
            catch
            end   
        end

        % Button pushed function: uartCloseButton
        function uartCloseButtonPushed(app, event)
            %close UART
            try
                delete(app.SERIAL);     %delete serialport object
            catch 
            end  
        end
    end

    % Component initialization
    methods (Access = private)

        % Create UIFigure and components
        function createComponents(app)

            % Create UIFigure and hide until all components are created
            app.UIFigure = uifigure('Visible', 'off');
            app.UIFigure.Position = [100 100 775 489];
            app.UIFigure.Name = 'MATLAB App';

            % Create dataPanel
            app.dataPanel = uipanel(app.UIFigure);
            app.dataPanel.Title = '数据显示';
            app.dataPanel.Position = [1 114 776 376];

            % Create originDataAxes
            app.originDataAxes = uiaxes(app.dataPanel);
            title(app.originDataAxes, '原始数据')
            xlabel(app.originDataAxes, 'X')
            ylabel(app.originDataAxes, 'Y')
            app.originDataAxes.TitleFontWeight = 'bold';
            app.originDataAxes.Position = [19 41 344 298];

            % Create fftDataAxes
            app.fftDataAxes = uiaxes(app.dataPanel);
            title(app.fftDataAxes, '距离维')
            xlabel(app.fftDataAxes, 'X')
            ylabel(app.fftDataAxes, 'Y')
            app.fftDataAxes.TitleFontWeight = 'bold';
            app.fftDataAxes.Position = [398 41 344 298];

            % Create uartPanel
            app.uartPanel = uipanel(app.UIFigure);
            app.uartPanel.Title = '端口设置';
            app.uartPanel.Position = [1 1 776 114];

            % Create Label
            app.Label = uilabel(app.uartPanel);
            app.Label.HorizontalAlignment = 'right';
            app.Label.Position = [30 58 41 22];
            app.Label.Text = '端口号';

            % Create uartPort_DropDown
            app.uartPort_DropDown = uidropdown(app.uartPanel);
            app.uartPort_DropDown.Items = {};
            app.uartPort_DropDown.Position = [86 58 277 22];
            app.uartPort_DropDown.Value = {};

            % Create Label_2
            app.Label_2 = uilabel(app.uartPanel);
            app.Label_2.HorizontalAlignment = 'right';
            app.Label_2.Position = [30 20 41 22];
            app.Label_2.Text = '波特率';

            % Create bandRate_DropDown
            app.bandRate_DropDown = uidropdown(app.uartPanel);
            app.bandRate_DropDown.Items = {};
            app.bandRate_DropDown.Position = [86 20 277 22];
            app.bandRate_DropDown.Value = {};

            % Create uartOpenButton
            app.uartOpenButton = uibutton(app.uartPanel, 'push');
            app.uartOpenButton.ButtonPushedFcn = createCallbackFcn(app, @uartOpenButtonPushed, true);
            app.uartOpenButton.Position = [437 56 306 24];
            app.uartOpenButton.Text = '打开串口';

            % Create uartCloseButton
            app.uartCloseButton = uibutton(app.uartPanel, 'push');
            app.uartCloseButton.ButtonPushedFcn = createCallbackFcn(app, @uartCloseButtonPushed, true);
            app.uartCloseButton.Position = [436 18 306 24];
            app.uartCloseButton.Text = '关闭串口';

            % Show the figure after all components are created
            app.UIFigure.Visible = 'on';
        end
    end

    % App creation and deletion
    methods (Access = public)

        % Construct app
        function app = adcdata_app

            % Create UIFigure and components
            createComponents(app)

            % Register the app with App Designer
            registerApp(app, app.UIFigure)

            % Execute the startup function
            runStartupFcn(app, @startupFcn)

            if nargout == 0
                clear app
            end
        end

        % Code that executes before app deletion
        function delete(app)

            % Delete UIFigure when app is deleted
            delete(app.UIFigure)
        end
    end
end

代码的内容其实也很简单，首先是选择串口号和波特率，然后点击打开串口按钮，打开串口，进行数据的读取，在程序中我设置了一个消息头的标志为chirpFlag，用于标识消息头，然后在读取的时候将其直接略过就ok了，然后紧接着读取消息体，组合成复数形式，然后做fft，最后分别将原始数据和fft后的数据显示到坐标轴上。
